Working with a remote database

db <- dbConnect_scidb("airlines")
flights <- tbl(db, "flights")
carriers <- tbl(db, "carriers")
airports <- tbl(db, "airports")
flights %>%
  head(5)
  1. Of all destinations from MSP, which were the 10 most common in 2015?
flights %>%
  filter(origin == "MSP", year == 2015) %>%
  group_by(dest) %>%
  summarize(
    n = n()
  ) %>%
  arrange(desc(n)) %>%
  head(10)
  1. Which airport had the highest average arrival delay time in 2015? What about the lowest?
flights %>%
  group_by(dest) %>%
  summarize(
    avg_arr_delay = mean(arr_delay, na.rm = TRUE)
  ) %>%
  arrange(desc(avg_arr_delay)) %>%
  head(10)
  1. How many flights were there in 2015 between MSP and FAI (Fairbanks International Airport?)
flights %>%
  filter(year == 2015) %>%
  filter((dest == "FAI" & origin == "MSP") | (origin == "FAI" & dest == "MSP")) %>%
  summarize(
    n = n()
  )
  1. List the airline and flight number for all flights between MSP and FAI in September 2015
flights %>%
  filter(year == 2015, month == 9) %>%
  filter((dest == "FAI" & origin == "MSP") | (origin == "FAI" & dest == "MSP")) %>%
  select(carrier, flight)

GROUP BY dest
ORDER BY numFlights DESC
LIMIT 0,10")

  1. How many flights were there in 2015 between MSP and FAI (Fairbanks International Airport?)
dbGetQuery(db,

  1. List the airline and flight number for all flights between MSP and FAI in September 2015

"SELECT carrier, flight 
FROM flights

")

LIMIT 0, 6;

LIMIT 0, 6;
